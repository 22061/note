#include <SoftwareSerial.h>

SoftwareSerial IM920Serial(8, 9);    // RX, TX

char input[30];
int i = 0;

void setup() {
  IM920Serial.begin(19200);
  Serial.begin(19200);
  IM920Serial.println("TXDU0002,hello");
}

void loop() {
  if (IM920Serial.available()) 
  {
    input[i] = IM920Serial.read();
    if (input[i] == '\n')
    {
      IM920Serial.println("TXDU0002,hello");
      Serial.println("hello");
      i = 0;
    }
    else
    {
        i++;
    }
  }
}
